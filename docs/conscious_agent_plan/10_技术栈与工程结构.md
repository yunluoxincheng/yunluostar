- CLI-first 本地运行形态

- sqlite-vec
- OpenAI / Claude / Qwen / Llama 等 LLM API

- LanceDB / Chroma / Qdrant / Milvus

## 4. Agent Controller 伪代码

```ts
export async function handleUserInput(userInput: string, sessionId: string) {
  const workingMemory = await loadWorkingMemory(sessionId);

  const intent = await inputInterpreter.parse(userInput);
  workingMemory.updateFromIntent(intent);

  const recalledMemories = await memoryAwakener.retrieve({
    userInput,
    workingMemory,
  });

  const cognitiveState = await cognitiveIntegrator.integrate({
    workingMemory,
    memories: recalledMemories,
    selfModel: await loadSelfModel(),
    userModel: await loadUserModel(),
  });

  const goal = await goalManager.selectGoal(cognitiveState);
  const plan = await planner.createPlan(goal, cognitiveState);

  const metacognitionResult = await monitor.check(plan, cognitiveState);
  const response = await llm.generateResponse(plan, cognitiveState, metacognitionResult);

  const episode = await episodeRecorder.record(userInput, response, cognitiveState);
  const reflection = await reflector.reflect(episode);
  await memoryConsolidator.consolidate(episode, reflection);

  await saveWorkingMemory(workingMemory);
  return response;
}
```
