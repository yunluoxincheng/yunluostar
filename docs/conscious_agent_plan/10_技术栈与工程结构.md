## 4. Agent Controller 伪代码

```python
async def handle_user_input(user_input: str, session_id: str):
    working_memory = load_working_memory(session_id)

    intent = input_interpreter.parse(user_input)
    working_memory.update_from_intent(intent)

    recalled_memories = memory_awakener.retrieve(
        user_input=user_input,
        working_memory=working_memory
    )

    cognitive_state = cognitive_integrator.integrate(
        working_memory=working_memory,
        memories=recalled_memories,
        self_model=load_self_model(),
        user_model=load_user_model()
    )

    goal = goal_manager.select_goal(cognitive_state)
    plan = planner.create_plan(goal, cognitive_state)

    metacognition_result = monitor.check(plan, cognitive_state)
    response = llm.generate_response(plan, cognitive_state, metacognition_result)

    episode = episode_recorder.record(user_input, response, cognitive_state)
    reflection = reflector.reflect(episode)
    memory_consolidator.consolidate(episode, reflection)

    save_working_memory(working_memory)
    return response
```
